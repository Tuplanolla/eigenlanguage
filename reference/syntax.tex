[\section*{\huge Syntax of Eigenlanguage}]

\section*{Primitive Forms}

\subsection*{Singleton}
\begin{lstlisting}[mathescape]
()
\end{lstlisting}

\subsection*{Function Application}
% f a_1 ...
\begin{lstlisting}[mathescape]
$f$ $a_1$ ...
\end{lstlisting}

\subsection*{Grouping}
% (x_1 ...)
\begin{lstlisting}[mathescape]
($x_1$ ...)
\end{lstlisting}

\subsection*{List Construction}
% [x_1 ...]
\begin{lstlisting}[mathescape]
[$x_1$ ...]
\end{lstlisting}

\subsection*{Function Definition}
% -> [p_1 ...] z
\begin{lstlisting}[mathescape]
-> [$p_1$ ...] $z$
\end{lstlisting}

\subsection*{Binding}
% = {y_1 x_1
%    ... ...}
%   z
\begin{lstlisting}[mathescape]
= {$y_1$ $x_1$
   ... ...}
  $z$
\end{lstlisting}

\subsection*{Module Declaration}
% <-> (m p_1 ...) {
% -> [e_1
%     (= [b_2 e_2
%         b_3 e_3
%         ... ...])
%     ...]
% <- [i_1
%     (i_2 a_2_1 ...)
%     (= [c_3 i_3
%         c_4 (i_4 a_4_1 ...)
%         ... ...])
%     ...]
% y_1 x_1
% ... ...
% }
\begin{lstlisting}[mathescape]
<-> ($m$ $p_1$ ...) {
-> [$e_1$
   (= [$b_2$ $e_2$
       $b_3$ $e_3$
       ... ...])
   ...]
<- [$i_1$
   ($i_2$ $a_{2, 1}$ ...)
   (= [$c_3$ $i_3$
       $c_4$ ($i_4$ $a_{4, 1}$ ...)
       ... ...])
   ...]
$y_1$ $x_1$
... ...
}
\end{lstlisting}

\columnbreak

\section*{Syntactic Sugar}

\subsection*{Name Qualification}
% m/y
\begin{lstlisting}[mathescape]
$m$/$y$
\end{lstlisting}

\subsection*{Number}
\begin{lstlisting}[mathescape]
20
\end{lstlisting}

\subsection*{Character}
% 'T'
\begin{lstlisting}[mathescape]
`T'
\end{lstlisting}

\subsection*{String}
% "This text
% is arbitrary."
\begin{lstlisting}[mathescape]
``This text
is arbitrary.''
\end{lstlisting}

\subsection*{Syntactic Comment}
% %x
\begin{lstlisting}[mathescape]
%$x$
\end{lstlisting}

\subsection*{Line Comment}
\begin{lstlisting}[mathescape]
% This text is arbitrary.
\end{lstlisting}

\subsection*{Block Comment}
\begin{lstlisting}[mathescape]
%%
This text
is arbitrary.
%%
\end{lstlisting}

\subsection*{Code as Data}
% `x
\begin{lstlisting}[mathescape]
${}^{\scriptscriptstyle\backslash}$$x$
\end{lstlisting}

\subsection*{Data as Code}
% ,x
\begin{lstlisting}[mathescape]
${}_{\scriptscriptstyle/}$$x$
\end{lstlisting}
